\newcommand*{\sign}[2]{  % insert signature
    \vspace*{2\baselineskip}
    \noindent
    \begin{minipage}{0.5\textwidth}
      \begin{flushleft}
        \SingleSpacing
            #1
       \end{flushleft}
    \end{minipage}
    \begin{minipage}{0.48\textwidth}
        \begin{flushright}
            \SingleSpacing
            #2
        \end{flushright}
    \end{minipage}
}

\newcommand*{\header}[2]{
    \begin{minipage}{0.4\textwidth}
      \begin{flushleft}
            #1
      \end{flushleft}
    \end{minipage}
    \begin{minipage}{0.6\textwidth}  % the field `to whom`
                                     % for example:

        \SingleSpacing
            #2
       \end{flushleft}
    \end{minipage}
    \vspace*{2\baselineskip}
}
